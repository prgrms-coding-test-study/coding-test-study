## 💭 아이디어
- deque 기반 너비 우선 탐색 활용
- 연결 요소마다 격자 칸의 숫자 누적 -> 계산 결과가 자연수인 경우에만 answer 리스트에 추가

## 🧑‍💻 코드
<!-- 작성한 코드를 백틱으로 감싸 넣어주세요 --> 
```python
from collections import deque

SEA = 'X'
NONE = -1

DIR = [[-1, 0], [1, 0], [0, -1], [0, 1]]

def solution(maps):
    row = len(maps)
    col = len(maps[0])
    visited = [[False] * col for _ in range(row)]
    answer = []

    for i in range(row):
        for j in range(col):
            if maps[i][j] != SEA and not visited[i][j]:
                total = visit(i, j, row, col, maps, visited)
                
                if total > 0:
                    answer.append(total)
    
    if answer:
        answer.sort()
    else:
        answer.append(NONE)
        
    return answer

def visit(x, y, row, col, maps, visited):
    queue = deque()
    
    queue.append([x, y])
    visited[x][y] = True
    total = int(maps[x][y])
    
    while queue:
        x, y = queue.popleft()
        
        for dx, dy in DIR:
            nx = x + dx
            ny = y + dy
            
            if nx < 0 or nx >= row or ny < 0 or ny >= col:
                continue
            if visited[nx][ny] or maps[nx][ny] == SEA:
                continue
            
            queue.append([nx, ny])
            visited[nx][ny] = True
            total += int(maps[nx][ny])
            
    return total
```

- 시간 복잡도: O(R * C)
- 공간 복잡도: O(R * C)
